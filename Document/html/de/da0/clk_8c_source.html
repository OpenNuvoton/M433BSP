273uint32_t CLK_SetCoreClock(uint32_t u32Hclk)
274{
275 uint32_t u32HIRCSTB;
276
277 /* Read HIRC clock source stable flag */
278 u32HIRCSTB = CLK->STATUS & CLK_STATUS_HIRCSTB_Msk;
279
280 /* The range of u32Hclk is running up to 144 MHz */
281 if(u32Hclk > FREQ_144MHZ)
282 {
283 u32Hclk = FREQ_144MHZ;
284 }
285
286 /* Switch HCLK clock source to HIRC clock for safe */
287 CLK->PWRCTL |= CLK_PWRCTL_HIRCEN_Msk;
288 CLK_WaitClockReady(CLK_STATUS_HIRCSTB_Msk);
289 CLK->CLKSEL0 |= CLK_CLKSEL0_HCLKSEL_Msk;
290 CLK->CLKDIV0 &= (~CLK_CLKDIV0_HCLKDIV_Msk);
291
292 /* Configure PLL setting if HXT clock is enabled */
293 if((CLK->PWRCTL & CLK_PWRCTL_HXTEN_Msk) == CLK_PWRCTL_HXTEN_Msk)
294 {
295 u32Hclk = CLK_EnablePLL(CLK_PLLCTL_PLLSRC_HXT, u32Hclk);
296 }
297 /* Configure PLL setting if HXT clock is not enabled */
298 else
299 {
300 u32Hclk = CLK_EnablePLL(CLK_PLLCTL_PLLSRC_HIRC, u32Hclk);
301
302 /* Read HIRC clock source stable flag */
303 u32HIRCSTB = CLK->STATUS & CLK_STATUS_HIRCSTB_Msk;
304 }
305
306 /* Select HCLK clock source to PLL,
307 and update system core clock
308 */
309 CLK_SetHCLK(CLK_CLKSEL0_HCLKSEL_PLL, CLK_CLKDIV0_HCLK(1UL));
310
311 /* Disable HIRC if HIRC is disabled before setting core clock */
312 if(u32HIRCSTB == 0UL)
313 {
314 CLK->PWRCTL &= ~CLK_PWRCTL_HIRCEN_Msk;
315 }
316
317 /* Return actually HCLK frequency is PLL frequency divide 1 */
318 return u32Hclk;
319}

335void CLK_SetHCLK(uint32_t u32ClkSrc, uint32_t u32ClkDiv)
336{
337 uint32_t u32HIRCSTB;
338
339 /* Read HIRC clock source stable flag */
340 u32HIRCSTB = CLK->STATUS & CLK_STATUS_HIRCSTB_Msk;
341
342 /* Switch to HIRC for Safe. Avoid HCLK too high when applying new divider. */
343 CLK->PWRCTL |= CLK_PWRCTL_HIRCEN_Msk;
344 CLK_WaitClockReady(CLK_STATUS_HIRCSTB_Msk);

349
350 /* Switch HCLK to new HCLK source */

352
353 /* Update System Core Clock */
354 SystemCoreClockUpdate();
355
356 /* Disable HIRC if HIRC is disabled before switching HCLK source */
357 if(u32HIRCSTB == 0UL)
358 {
359 CLK->PWRCTL &= ~CLK_PWRCTL_HIRCEN_Msk;

762uint32_t CLK_EnablePLL(uint32_t u32PllClkSrc, uint32_t u32PllFreq)
763{
764 uint32_t u32PllSrcClk, u32NR, u32NF, u32NO, u32CLK_SRC, u32PllClk;
765 uint32_t u32Tmp, u32Tmp2, u32Tmp3, u32Min, u32MinNF, u32MinNR, u32MinNO, u32basFreq;
766
767 /* Disable PLL first to avoid unstable when setting PLL */
768 CLK_DisablePLL();
769
770 /* PLL source clock is from HXT */
771 if(u32PllClkSrc == CLK_PLLCTL_PLLSRC_HXT)
772 {
773 /* Enable HXT clock */
774 CLK->PWRCTL |= CLK_PWRCTL_HXTEN_Msk;
775
776 /* Wait for HXT clock ready */
777 CLK_WaitClockReady(CLK_STATUS_HXTSTB_Msk);
778
779 /* Select PLL source clock from HXT */
780 u32CLK_SRC = CLK_PLLCTL_PLLSRC_HXT;
781 u32PllSrcClk = __HXT;
782
783 /* Check HXT range. Set u32NR start value */
784 /* Constraint 1: 4MHz <= FREF <= 8MHz */
785 if((u32PllSrcClk >= 4000000UL) && (u32PllSrcClk <= 8000000UL))
786 u32NR = 1UL;
787 else
788 u32NR = 2UL;
789 }
790
791 /* PLL source clock is from HIRC */
792 else
793 {
794 /* Enable HIRC clock */
795 CLK->PWRCTL |= CLK_PWRCTL_HIRCEN_Msk;
796
797 /* Wait for HIRC clock ready */
798 CLK_WaitClockReady(CLK_STATUS_HIRCSTB_Msk);
799
800 /* Select PLL source clock from HIRC */
801 u32CLK_SRC = CLK_PLLCTL_PLLSRC_HIRC;
802 u32PllSrcClk = __HIRC;
803
804 /* u32NR start from 4 when FIN = 22.1184MHz to avoid calculation overflow */
805 u32NR = 4UL;
806 }
807
808 if((u32PllFreq <= FREQ_500MHZ) && (u32PllFreq >= FREQ_50MHZ))
809 {
810
811 /* Find best solution */
812 u32Min = (uint32_t) - 1;
813 u32MinNR = 0UL;
814 u32MinNF = 0UL;
815 u32MinNO = 0UL;
816 u32basFreq = u32PllFreq;
817
818 for(u32NO = 1UL; u32NO <= 4UL; u32NO++)
819 {
820 /* Break when get good results */
821 if (u32Min == 0UL)
822 {
823 break;
824 }
825
826 if (u32NO != 3UL)
827 {
828
829 if(u32NO == 4UL)
830 {
831 u32PllFreq = u32basFreq << 2;
832 }
833 else if(u32NO == 2UL)
834 {
835 u32PllFreq = u32basFreq << 1;
836 }
837 else
838 {
839 }
840
841 for(u32NR = 1UL; u32NR <= 32UL; u32NR++)
842 {
843 /* Break when get good results */
844 if (u32Min == 0UL)
845 {
846 break;
847 }
848
849 u32Tmp = u32PllSrcClk / u32NR;
850 if((u32Tmp >= 4000000UL) && (u32Tmp <= 8000000UL))
851 {
852 for(u32NF = 2UL; u32NF <= 513UL; u32NF++)
853 {
854 /* u32Tmp2 is shifted 2 bits to avoid overflow */
855 u32Tmp2 = (((u32Tmp * 2UL) >> 2) * u32NF);
856
857 if((u32Tmp2 >= FREQ_50MHZ) && (u32Tmp2 <= FREQ_125MHZ))
858 {
859 u32Tmp3 = (u32Tmp2 > (u32PllFreq>>2)) ? u32Tmp2 - (u32PllFreq>>2) : (u32PllFreq>>2) - u32Tmp2;
860 if(u32Tmp3 < u32Min)
861 {
862 u32Min = u32Tmp3;
863 u32MinNR = u32NR;
864 u32MinNF = u32NF;
865 u32MinNO = u32NO;
866
867 /* Break when get good results */
868 if(u32Min == 0UL)
869 {
870 break;
871 }
872 }
873 }
874 }
875 }
876 }
877 }
878 }
879
880 /* Enable and apply new PLL setting. */
881 CLK->PLLCTL = u32CLK_SRC | ((u32MinNO - 1UL) << 14) | ((u32MinNR - 1UL) << 9) | (u32MinNF - 2UL);
882
883 /* Wait for PLL clock stable */
884 CLK_WaitClockReady(CLK_STATUS_PLLSTB_Msk);
885
886 /* Actual PLL output clock frequency */
887 u32PllClk = u32PllSrcClk / (u32MinNO * (u32MinNR)) * (u32MinNF) * 2UL;
888 }
889 else
890 {
891 /* Wrong frequency request. Just return default setting. */
892 /* Apply default PLL setting and return */
893 if(u32PllClkSrc == CLK_PLLCTL_PLLSRC_HXT)
894 {
895 CLK->PLLCTL = CLK_PLLCTL_144MHz_HXT;
896 }
897 else
898 {
899 CLK->PLLCTL = CLK_PLLCTL_144MHz_HIRC;
900 }
901
902 /* Wait for PLL clock stable */
903 CLK_WaitClockReady(CLK_STATUS_PLLSTB_Msk);
904
905 /* Actual PLL output clock frequency */
906 u32PllClk = CLK_GetPLLClockFreq();
907 }
908
909 return u32PllClk;
910}

1148uint32_t CLK_GetPLLClockFreq(void)
1149{
1150 uint32_t u32PllFreq = 0UL, u32PllReg;
1151 uint32_t u32FIN, u32NF, u32NR, u32NO;
1152 uint8_t au8NoTbl[4] = {1U, 2U, 2U, 4U};
1153
1154 u32PllReg = CLK->PLLCTL;
1155
1156 if(u32PllReg & (CLK_PLLCTL_PD_Msk | CLK_PLLCTL_OE_Msk))
1157 {
1158 u32PllFreq = 0UL; /* PLL is in power down mode or fix low */
1159 }
1160 else if((u32PllReg & CLK_PLLCTL_BP_Msk) == CLK_PLLCTL_BP_Msk)
1161 {
1162 if((u32PllReg & CLK_PLLCTL_PLLSRC_HIRC) == CLK_PLLCTL_PLLSRC_HIRC)
1163 {
1164 u32FIN = __HIRC; /* PLL source clock from HIRC */
1165 }
1166 else
1167 {
1168 u32FIN = __HXT; /* PLL source clock from HXT */
1169 }
1170
1171 u32PllFreq = u32FIN;
1172 }
1173 else
1174 {
1175 if((u32PllReg & CLK_PLLCTL_PLLSRC_HIRC) == CLK_PLLCTL_PLLSRC_HIRC)
1176 {
1177 u32FIN = __HIRC; /* PLL source clock from HIRC */
1178 }
1179 else
1180 {
1181 u32FIN = __HXT; /* PLL source clock from HXT */
1182 }
1183 /* PLL is output enabled in normal work mode */
