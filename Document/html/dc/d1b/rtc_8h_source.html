192__STATIC_INLINE void RTC_WaitAccessEnable(void)
193{
194 uint32_t u32TimeOutCount = SystemCoreClock; // 1 second timeout
195 uint32_t i = 0;
196
197 while((RTC->RWEN & RTC_RWEN_RTCBUSY_Msk) == RTC_RWEN_RTCBUSY_Msk)
198 {
199 i++;
200 if(i > u32TimeOutCount) break;
201 }
202
203 /* To wait RWENF bit is cleared and enable RWENF bit (Access Enable bit) again */
204 RTC->RWEN = RTC_WRITE_KEY;
205
206 /* To wait RWENF bit is set and user can access the protected-register of RTC from now on */
207 while((RTC->RWEN & RTC_RWEN_RWENF_Msk) == (uint32_t)0x0)
208 {
209 i++;
210 if(i > u32TimeOutCount) break;
211 }
212}
