161__STATIC_INLINE uint32_t FMC_ReadCID(void)
162{
163 uint32_t tout = FMC_TIMEOUT_READ;
164
165 g_FMC_i32ErrCode = 0;
166 FMC->ISPCMD = FMC_ISPCMD_READ_CID; /* Set ISP Command Code */
167 FMC->ISPADDR = 0x0u; /* Must keep 0x0 when read CID */
168 FMC->ISPTRG = FMC_ISPTRG_ISPGO_Msk; /* Trigger to start ISP procedure */
169#if ISBEN
170 __ISB();
171#endif /* To make sure ISP/CPU be Synchronized */
172 while (tout-- > 0)
173 {
174 if (!(FMC->ISPTRG & FMC_ISPTRG_ISPGO_Msk)) /* Waiting for ISP Done */
175 {
176 if (FMC->ISPDAT != 0xDA)
177 g_FMC_i32ErrCode = -1;
178 return FMC->ISPDAT;
179 }
180 }
181 g_FMC_i32ErrCode = -1;
182 return 0xFFFFFFFF;
183}
