478{
479 int32_t rev = 1l;
480
481 if((tCAN->IF[1].MCON & CAN_IF_MCON_NEWDAT_Msk) == 0ul)
482 {
483 /* In basic mode, receive data always save in IF2 */
484 rev = 0; /* return FALSE */
485 }
486 else
487 {
488
489 tCAN->STATUS &= (~CAN_STATUS_RXOK_Msk);
490
491 tCAN->IF[1].CMASK = CAN_IF_CMASK_ARB_Msk
492 | CAN_IF_CMASK_CONTROL_Msk
493 | CAN_IF_CMASK_DATAA_Msk
494 | CAN_IF_CMASK_DATAB_Msk;
495
496 if((tCAN->IF[1].ARB2 & CAN_IF_ARB2_XTD_Msk) == 0ul)

500 pCanMsg->Id = (tCAN->IF[1].ARB2 >> 2) & 0x07FFul;
501
502 }
503 else
504 {
505 /* extended ID*/
506 pCanMsg->IdType = CAN_EXT_ID;
507 pCanMsg->Id = (tCAN->IF[1].ARB2 & 0x1FFFul) << 16;
508 pCanMsg->Id |= (uint32_t)tCAN->IF[1].ARB1;
509 }
510

512
513 pCanMsg->DLC = (uint8_t)(tCAN->IF[1].MCON & CAN_IF_MCON_DLC_Msk);
514 pCanMsg->Data[0] = (uint8_t)(tCAN->IF[1].DAT_A1 & CAN_IF_DAT_A1_DATA0_Msk);

683int32_t CAN_ReadMsgObj(CAN_T *tCAN, uint8_t u8MsgObj, uint8_t u8Release, STR_CANMSG_T* pCanMsg)
684{
685 int32_t rev = 1l;
686 uint32_t u32MsgIfNum;
687 uint32_t u32TimeOutCount = SystemCoreClock * 2; // 2 second timeout
688
689 if(!CAN_IsNewDataReceived(tCAN, u8MsgObj))
690 {
691 rev = 0; /* return FALSE */
692 }
693 else
694 {
695 /* Get and lock a free interface */
696 if((u32MsgIfNum = LockIF_TL(tCAN)) == 2ul)
697 {
698 rev = 0; /* return FALSE */
699 }
700 else
701 {
702 tCAN->STATUS &= (~CAN_STATUS_RXOK_Msk);
703
704 /* read the message contents*/
705 tCAN->IF[u32MsgIfNum].CMASK = CAN_IF_CMASK_MASK_Msk
706 | CAN_IF_CMASK_ARB_Msk
707 | CAN_IF_CMASK_CONTROL_Msk
708 | CAN_IF_CMASK_CLRINTPND_Msk
709 | (u8Release ? CAN_IF_CMASK_TXRQSTNEWDAT_Msk : 0ul)
710 | CAN_IF_CMASK_DATAA_Msk
711 | CAN_IF_CMASK_DATAB_Msk;
712
713 tCAN->IF[u32MsgIfNum].CREQ = 1ul + u8MsgObj;
714
715 while(tCAN->IF[u32MsgIfNum].CREQ & CAN_IF_CREQ_BUSY_Msk) /*Wait*/
716 {
717 if(u32TimeOutCount == 0)
718 {
719 ReleaseIF(tCAN, u32MsgIfNum);
720 return -1;
721 }
722 u32TimeOutCount--;
723 }
724
725 if((tCAN->IF[u32MsgIfNum].ARB2 & CAN_IF_ARB2_XTD_Msk) == 0ul)
726 {
727 /* standard ID*/
728 pCanMsg->IdType = CAN_STD_ID;

737
738 pCanMsg->DLC = (uint8_t)(tCAN->IF[u32MsgIfNum].MCON & CAN_IF_MCON_DLC_Msk);
739 pCanMsg->Data[0] = (uint8_t)(tCAN->IF[u32MsgIfNum].DAT_A1 & CAN_IF_DAT_A1_DATA0_Msk);
740 pCanMsg->Data[1] = (uint8_t)((tCAN->IF[u32MsgIfNum].DAT_A1 & CAN_IF_DAT_A1_DATA1_Msk) >> CAN_IF_DAT_A1_DATA1_Pos);
741 pCanMsg->Data[2] = (uint8_t)(tCAN->IF[u32MsgIfNum].DAT_A2 & CAN_IF_DAT_A2_DATA2_Msk);

785 {
786 u64PCLK_DIV = (uint64_t)((CLK->PCLKDIV & CLK_PCLKDIV_APB1DIV_Msk) >> CLK_PCLKDIV_APB1DIV_Pos);
787 u64PCLK_DIV = (uint64_t)(1 << u64PCLK_DIV);
788 }
789
790 clock_freq = SystemCoreClock / u64PCLK_DIV;
791
792 if(u32BaudRate >= (uint32_t)1000000)
793 {
794 u32BaudRate = (uint32_t)1000000;
795 }
796
797 /* Use CIA recommended sample points */
798 if (u32BaudRate > (uint32_t)800000)
799 {
800 sampl_pt = (int)750;
801 }
802 else if (u32BaudRate > (uint32_t)500000)
803 {
804 sampl_pt = (int)800;
805 }
806 else
807 {
808 sampl_pt = (int)875;
809 }
810
811 /* tseg even = round down, odd = round up */
812 for (tseg = (TSEG1_MAX + TSEG2_MAX) * 2ul + 1ul; tseg >= (TSEG1_MIN + TSEG2_MIN) * 2ul; tseg--)
813 {
814 tsegall = 1ul + tseg / 2ul;
815 /* Compute all possible tseg choices (tseg=tseg1+tseg2) */
816 brp = clock_freq / (tsegall * u32BaudRate) + tseg % 2;
817 /* chose brp step which is possible in system */
818 brp = (brp / BRP_INC) * BRP_INC;
819
820 if ((brp < BRP_MIN) || (brp > BRP_MAX))
821 {
822 continue;
823 }
824 rate = clock_freq / (brp * tsegall);
825
826 error = u32BaudRate - rate;
827
828 /* tseg brp biterror */
829 if (error < 0)
830 {
831 error = -error;
832 }
833 if (error > best_error)
834 {
835 continue;
836 }
837 best_error = error;
838 if (error == 0)
839 {
840 spt = can_update_spt(sampl_pt, tseg / 2, &tseg1, &tseg2);
841 error = sampl_pt - spt;
842 if (error < 0)
843 {
844 error = -error;
845 }
846 if (error > spt_error)
847 {
848 continue;
849 }
850 spt_error = error;
851 }
852 best_tseg = tseg / 2;
853 best_brp = brp;
854
855 if (error == 0)
856 {
857 break;
858 }
859 }
860
861 spt = can_update_spt(sampl_pt, best_tseg, &tseg1, &tseg2);
862
863 /* check for sjw user settings */
864 /* bt->sjw is at least 1 -> sanitize upper bound to sjw_max */
865 if (sjw > SJW_MAX)
866 {
867 sjw = SJW_MAX;
868 }
869 /* bt->sjw must not be higher than tseg2 */
870 if (tseg2 < sjw)
871 {
872 sjw = tseg2;
873 }
874
875 /* real bit-rate */
876 u32BaudRate = clock_freq / (best_brp * (tseg1 + tseg2 + 1));
877
878 tCAN->BTIME = ((uint32_t)(tseg2 - 1ul) << CAN_BTIME_TSEG2_Pos) | ((uint32_t)(tseg1 - 1ul) << CAN_BTIME_TSEG1_Pos) |
879 ((uint32_t)(best_brp - 1ul) & CAN_BTIME_BRP_Msk) | (sjw << CAN_BTIME_SJW_Pos);
880 tCAN->BRPE = ((uint32_t)(best_brp - 1ul) >> 6) & 0x0Ful;
881
882 /* printf("\n bitrate = %d \n", CAN_GetCANBitRate(tCAN)); */
883
884 CAN_LeaveInitMode(tCAN);
885
886 return u32BaudRate;
887}
