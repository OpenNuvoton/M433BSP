381int32_t FMC_WriteMultiple(uint32_t u32Addr, uint32_t pu32Buf[], uint32_t u32Len)
382{
383 int i, idx, retval = 0;
384 int32_t tout;
385
386 g_FMC_i32ErrCode = 0;
387
388 if ((u32Addr >= FMC_APROM_END) || ((u32Addr % 8) != 0))
389 {
390 g_FMC_i32ErrCode = -2;
391 return -2;
392 }
393
394 u32Len = u32Len - (u32Len % 8); /* u32Len must be multiple of 8. */
395
396 idx = 0;
397
398 while (u32Len >= 8)
399 {
400 FMC->ISPADDR = u32Addr;
401 FMC->MPDAT0 = pu32Buf[idx++];
402 FMC->MPDAT1 = pu32Buf[idx++];
403 FMC->MPDAT2 = pu32Buf[idx++];
404 FMC->MPDAT3 = pu32Buf[idx++];
405 FMC->ISPCMD = FMC_ISPCMD_PROGRAM_MUL;
406 FMC->ISPTRG = FMC_ISPTRG_ISPGO_Msk;
407 u32Len -= 16;
408 retval += 16;
409
410 for (i = 16; i < FMC_MULTI_WORD_PROG_LEN; i += 16)
411 {
412 tout = FMC_TIMEOUT_WRITE;
413 while ((tout-- > 0) && (FMC->MPSTS & (FMC_MPSTS_D0_Msk | FMC_MPSTS_D1_Msk))) {}
414 if (tout <= 0)
415 {
416 g_FMC_i32ErrCode = -1;
417 return -1;
418 }
419 if (!(FMC->MPSTS & FMC_MPSTS_MPBUSY_Msk))
420 {
421 /* printf(" [WARNING] busy cleared after D0D1 cleared!\n"); */
422 break;
423 }
424
425 if (u32Len < 8)
426 break;
427
428 FMC->MPDAT0 = pu32Buf[idx++];
429 FMC->MPDAT1 = pu32Buf[idx++];
430 retval += 8;
431 u32Len -= 8;
432
433 tout = FMC_TIMEOUT_WRITE;
434 while ((tout-- > 0) && (FMC->MPSTS & (FMC_MPSTS_D2_Msk | FMC_MPSTS_D3_Msk))) {}
435 if (tout <= 0)
436 {
437 g_FMC_i32ErrCode = -1;
438 return -1;
439 }
440 if (!(FMC->MPSTS & FMC_MPSTS_MPBUSY_Msk))
441 {
442 /* printf(" [WARNING] busy cleared after D2D3 cleared!\n"); */
443 i += 8;
444 break;
445 }
446
447 if (u32Len < 8)
448 break;
449
450 FMC->MPDAT2 = pu32Buf[idx++];
451 FMC->MPDAT3 = pu32Buf[idx++];
452 retval += 8;
453 u32Len -= 8;
454 }
455
456 if (i != FMC_MULTI_WORD_PROG_LEN)
457 {
458 /* printf(" [WARNING] Multi-word program interrupted at 0x%x !!\n", i); */
459 return retval;
460 }
461
462 tout = FMC_TIMEOUT_WRITE;
463 while ((tout-- > 0) && (FMC->MPSTS & FMC_MPSTS_MPBUSY_Msk)) {}
464 if (tout <= 0)
465 {
466 g_FMC_i32ErrCode = -1;
467 return -1;
468 }
469
470 u32Addr += FMC_MULTI_WORD_PROG_LEN;
471 }
472 return retval;
473}

916int32_t FMC_SetSPKey(uint32_t key[3], uint32_t kpmax, uint32_t kemax,
917 const int32_t lock_CONFIG)
918{
919 uint32_t lock_ctrl = 0UL;
920 uint32_t u32KeySts;
921 int32_t tout;
922 int32_t ret = 0;
923
924 g_FMC_i32ErrCode = 0;
925
926 if (FMC->KPKEYSTS != 0x200UL)
927 {
928 ret = -1;
929 }
930
931 if (FMC_Erase(FMC_KPROM_BASE))
932 {
933 ret = -2;
934 }
935
936 if (FMC_Erase(FMC_KPROM_BASE+0x200UL))
937 {
938 ret = -3;
939 }
940
941 if (!lock_CONFIG)
942 {
943 lock_ctrl |= 0x1UL;
944 }
945
946 if (ret == 0)
947 {
948 FMC_Write(FMC_KPROM_BASE, key[0]);
949 FMC_Write(FMC_KPROM_BASE+0x4UL, key[1]);
950 FMC_Write(FMC_KPROM_BASE+0x8UL, key[2]);
951 FMC_Write(FMC_KPROM_BASE+0xCUL, kpmax);
952 FMC_Write(FMC_KPROM_BASE+0x10UL, kemax);
953 FMC_Write(FMC_KPROM_BASE+0x14UL, lock_ctrl);
954
955 tout = FMC_TIMEOUT_WRITE;
956 while ((tout-- > 0) && (FMC->KPKEYSTS & FMC_KPKEYSTS_KEYBUSY_Msk)) {}
957 if (tout <= 0)
958 {
959 g_FMC_i32ErrCode = -3;
960 return -3;
961 }
962
963 u32KeySts = FMC->KPKEYSTS;
964
965 if (!(u32KeySts & FMC_KPKEYSTS_KEYLOCK_Msk))
966 {
967 /* Security key lock failed! */
968 ret = -4;
969 }
970 else if ((lock_CONFIG && (!(u32KeySts & FMC_KPKEYSTS_CFGFLAG_Msk))) ||
971 ((!lock_CONFIG) && (u32KeySts & FMC_KPKEYSTS_CFGFLAG_Msk)))
972 {
973 /* CONFIG lock failed! */
974 ret = -5;
975 }
976 else if (((FMC->KPCNT & FMC_KPCNT_KPMAX_Msk) >> FMC_KPCNT_KPMAX_Pos) != kpmax)
977 {
978 /* KPMAX failed! */
979 ret = -7;
980 }
981 else if (((FMC->KPKEYCNT & FMC_KPKEYCNT_KPKEMAX_Msk) >> FMC_KPKEYCNT_KPKEMAX_Pos) != kemax)
982 {
983 /* KEMAX failed! */
984 ret = -8;
985 }
986 }
987 g_FMC_i32ErrCode = ret;
988 return ret;
989}
