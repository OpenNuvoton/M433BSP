116void RTC_32KCalibration(int32_t i32FrequencyX10000)
117{
118 int32_t i32RegInt,i32RegFra ;
119
120 /* Compute Integer and Fraction for RTC register*/
121 i32RegInt = (i32FrequencyX10000/10000) - 32752;
122 i32RegFra = ((((i32FrequencyX10000%10000)) * 64) + 5000) / 10000;
123
124 if(i32RegFra >= 0x40)
125 {
126 i32RegFra = 0x0;
127 i32RegInt++;
128 }
129
130 /* Judge Integer part is reasonable */
131 if ( (i32RegInt < 0) | (i32RegInt > 31) )
132 {
133 return;
134 }
135
136 RTC_WaitAccessEnable();
137 RTC->FREQADJ = (uint32_t)((i32RegInt<<8) | i32RegFra);
138}

158void RTC_GetDateAndTime(S_RTC_TIME_DATA_T *sPt)
159{
160 uint32_t u32Tmp;
161
162 sPt->u32TimeScale = RTC->CLKFMT & RTC_CLKFMT_24HEN_Msk; /* 12/24-hour */
163 sPt->u32DayOfWeek = RTC->WEEKDAY & RTC_WEEKDAY_WEEKDAY_Msk; /* Day of the week */
164
165 /* Get [Date digit] data */
166 g_u32hiYear = (RTC->CAL & RTC_CAL_TENYEAR_Msk) >> RTC_CAL_TENYEAR_Pos;
167 g_u32loYear = (RTC->CAL & RTC_CAL_YEAR_Msk) >> RTC_CAL_YEAR_Pos;
168 g_u32hiMonth = (RTC->CAL & RTC_CAL_TENMON_Msk) >> RTC_CAL_TENMON_Pos;
169 g_u32loMonth = (RTC->CAL & RTC_CAL_MON_Msk) >> RTC_CAL_MON_Pos;
170 g_u32hiDay = (RTC->CAL & RTC_CAL_TENDAY_Msk) >> RTC_CAL_TENDAY_Pos;
171 g_u32loDay = (RTC->CAL & RTC_CAL_DAY_Msk) >> RTC_CAL_DAY_Pos;
172
173 /* Get [Time digit] data */
174 g_u32hiHour = (RTC->TIME & RTC_TIME_TENHR_Msk) >> RTC_TIME_TENHR_Pos;
175 g_u32loHour = (RTC->TIME & RTC_TIME_HR_Msk) >> RTC_TIME_HR_Pos;
176 g_u32hiMin = (RTC->TIME & RTC_TIME_TENMIN_Msk) >> RTC_TIME_TENMIN_Pos;
177 g_u32loMin = (RTC->TIME & RTC_TIME_MIN_Msk) >> RTC_TIME_MIN_Pos;
178 g_u32hiSec = (RTC->TIME & RTC_TIME_TENSEC_Msk) >> RTC_TIME_TENSEC_Pos;
179 g_u32loSec = (RTC->TIME & RTC_TIME_SEC_Msk) >> RTC_TIME_SEC_Pos;
180
181 /* Compute to 20XX year */
182 u32Tmp = (g_u32hiYear * 10ul);
183 u32Tmp += g_u32loYear;
184 sPt->u32Year = u32Tmp + RTC_YEAR2000;
185
186 /* Compute 0~12 month */
187 u32Tmp = (g_u32hiMonth * 10ul);
188 sPt->u32Month = u32Tmp + g_u32loMonth;
189
190 /* Compute 0~31 day */
191 u32Tmp = (g_u32hiDay * 10ul);
192 sPt->u32Day = u32Tmp + g_u32loDay;
193
194 /* Compute 12/24 hour */
195 if(sPt->u32TimeScale == RTC_CLOCK_12)
196 {
197 u32Tmp = (g_u32hiHour * 10ul);
198 u32Tmp += g_u32loHour;
199 sPt->u32Hour = u32Tmp; /* AM: 1~12. PM: 21~32. */
200
201 if(sPt->u32Hour >= 21ul)
202 {
203 sPt->u32AmPm = RTC_PM;
204 sPt->u32Hour -= 20ul;
205 }
206 else
207 {
208 sPt->u32AmPm = RTC_AM;
209 }
210
211 u32Tmp = (g_u32hiMin * 10ul);
212 u32Tmp += g_u32loMin;
213 sPt->u32Minute = u32Tmp;
214
215 u32Tmp = (g_u32hiSec * 10ul);
216 u32Tmp += g_u32loSec;
217 sPt->u32Second = u32Tmp;
218 }
219 else
220 {
221 u32Tmp = (g_u32hiHour * 10ul);
222 u32Tmp += g_u32loHour;
223 sPt->u32Hour = u32Tmp;
224
225 u32Tmp = (g_u32hiMin * 10ul);
226 u32Tmp += g_u32loMin;
227 sPt->u32Minute = u32Tmp;
228
229 u32Tmp = (g_u32hiSec * 10ul);
230 u32Tmp += g_u32loSec;
231 sPt->u32Second = u32Tmp;

353void RTC_SetDateAndTime(S_RTC_TIME_DATA_T *sPt)
354{
355 uint32_t u32RegCAL, u32RegTIME;
356
357 if(sPt == 0ul)
358 {
359 }
360 else
361 {
362 /*-----------------------------------------------------------------------------------------------------*/
363 /* Set RTC 24/12 hour setting and Day of the Week */
364 /*-----------------------------------------------------------------------------------------------------*/
365 RTC_WaitAccessEnable();
366 if(sPt->u32TimeScale == RTC_CLOCK_12)
367 {
368 RTC->CLKFMT &= ~RTC_CLKFMT_24HEN_Msk;
369
370 /*-------------------------------------------------------------------------------------------------*/
371 /* Important, range of 12-hour PM mode is 21 up to 32 */
372 /*-------------------------------------------------------------------------------------------------*/
373 if(sPt->u32AmPm == RTC_PM)
374 {
375 sPt->u32Hour += 20ul;
376 }
377 }
378 else
379 {
380 RTC->CLKFMT |= RTC_CLKFMT_24HEN_Msk;
381 }
382
383 /* Set Day of the Week */
384 RTC_WaitAccessEnable();
385 RTC->WEEKDAY = sPt->u32DayOfWeek;
386
387 /*-----------------------------------------------------------------------------------------------------*/
388 /* Set RTC Current Date and Time */
389 /*-----------------------------------------------------------------------------------------------------*/
390 u32RegCAL = ((sPt->u32Year - RTC_YEAR2000) / 10ul) << 20;
391 u32RegCAL |= (((sPt->u32Year - RTC_YEAR2000) % 10ul) << 16);
392 u32RegCAL |= ((sPt->u32Month / 10ul) << 12);
393 u32RegCAL |= ((sPt->u32Month % 10ul) << 8);
394 u32RegCAL |= ((sPt->u32Day / 10ul) << 4);
395 u32RegCAL |= (sPt->u32Day % 10ul);
396
397 u32RegTIME = ((sPt->u32Hour / 10ul) << 20);
398 u32RegTIME |= ((sPt->u32Hour % 10ul) << 16);
399 u32RegTIME |= ((sPt->u32Minute / 10ul) << 12);
400 u32RegTIME |= ((sPt->u32Minute % 10ul) << 8);
401 u32RegTIME |= ((sPt->u32Second / 10ul) << 4);
402 u32RegTIME |= (sPt->u32Second % 10ul);
403
404 /*-----------------------------------------------------------------------------------------------------*/
405 /* Set RTC Calender and Time Loading */
406 /*-----------------------------------------------------------------------------------------------------*/
407 RTC_WaitAccessEnable();
408 RTC->CAL = (uint32_t)u32RegCAL;
409 RTC_WaitAccessEnable();
410 RTC->TIME = (uint32_t)u32RegTIME;
411 }
